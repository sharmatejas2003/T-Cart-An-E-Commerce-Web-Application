*****  MYSQL Schema *****

--- Users Table ---

create table users(
id int auto_increment primary key,
name varchar(100),
email varchar(100),
password varchar(100),
role varchar(30)
);




--- Products Table ---

create table products(
id int auto_increment primary key,
name varchar(100),
description varchar(255),
price double
);



--- Order Table ---

create table orders(
id int auto_increment primary key,
user_id int,
total double,
order_date date
);



--- Order Items Table ---

create table order_items(
id int auto_increment primary key,
order_id int,
product_id int,
quantity int
);





--- Category-of-Products Table ---

CREATE TABLE categories (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(100)
);
